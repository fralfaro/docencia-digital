Manipulación de datos con pandas

Del dato crudo al análisis en Google Colab

pandas es la librería central del ecosistema de ciencia de datos en Python. Si NumPy es la calculadora, pandas es la hoja de cálculo programable: permite cargar, explorar, limpiar, transformar y exportar datos tabulares con una sintaxis expresiva y eficiente.

Este capítulo cubre pandas desde la perspectiva docente: cómo cargar datos de distintas fuentes (incluyendo Google Drive y URLs), inspeccionar un dataset desconocido, aplicar las transformaciones más frecuentes, y producir un análisis reproducible completo. Cada sección incluye código listo para usar en Colab.

Cargar datos: CSV, Excel, URL, Google Drive

El DataFrame: la estructura central de pandas

Un DataFrame es una tabla bidimensional con filas y columnas etiquetadas. Cada columna es una Serie — un arreglo unidimensional con un índice. Esta estructura mapea directamente a lo que un docente o investigador tiene en una hoja de cálculo.

Cargar desde CSV

El formato CSV es el más común para compartir datos tabulares.

# CSV local (en el runtime de Colab o Drive montado)
df = pd.read_csv('datos.csv')

# Opciones frecuentes
df = pd.read_csv(
    'datos.csv',
    sep=';',                    # separador (punto y coma en archivos latinoamericanos)
    encoding='utf-8-sig',       # encoding para caracteres en español (ñ, tildes)
    decimal=',',                # coma decimal (Chile, España)
    thousands='.',              # separador de miles
    na_values=['N/A', '-', ''],  # valores que se interpretan como NaN
    parse_dates=['fecha'],       # columnas que se convierten a datetime
    index_col='id',             # columna que actúa como índice
    usecols=['nombre', 'nota', 'carrera'],  # solo cargar ciertas columnas
    nrows=1000                  # cargar solo las primeras N filas (útil para explorar datasets grandes)
)
TipCSV con punto y coma: el caso latinoamericano

Excel en sistemas configurados en español guarda los CSV con punto y coma como separador (porque la coma se usa como decimal). Si recibes un CSV y pd.read_csv('archivo.csv') muestra todo en una sola columna, prueba sep=';'. También puedes usar pd.read_csv('archivo.csv', sep=None, engine='python') para que pandas detecte el separador automáticamente.

Función de inspección rápida para clase

Esta función combina todo lo anterior en un solo output formateado — útil como primera celda de análisis en notebooks de laboratorio:

def inspeccionar(df, nombre="DataFrame"):
    """Inspección rápida estandarizada para notebooks de clase."""
    print(f"{'='*60}")
    print(f"  {nombre}")
    print(f"{'='*60}")
    print(f"\n📐 Dimensiones: {df.shape[0]:,} filas × {df.shape[1]} columnas")
    print(f"💾 Memoria: {df.memory_usage(deep=True).sum() / 1024:.1f} KB\n")

    print("📋 Columnas y tipos:")
    for col in df.columns:
        nulos = df[col].isnull().sum()
        pct   = nulos / len(df) * 100
        unicos = df[col].nunique()
        print(f"   {col:<25} {str(df[col].dtype):<15} "
              f"nulos: {nulos:4d} ({pct:5.1f}%)   únicos: {unicos}")

    print("\n📊 Estadísticas numéricas:")
    print(df.describe().round(3).to_string())
    print(f"\n{'='*60}\n")

# Uso
inspeccionar(df, nombre="Dataset notas MAT281 2025-1")

Filtros, selección de columnas y condiciones

Selección de columnas

# Una columna → Serie
notas = df['nota_final']

# Múltiples columnas → DataFrame
df_subset = df[['nombre', 'carrera', 'nota_final']]

# Selección por posición: iloc[filas, columnas]
df.iloc[0]              # primera fila
df.iloc[0:5]            # filas 0 a 4
df.iloc[:, 0:3]         # todas las filas, columnas 0 a 2
df.iloc[0:5, 1:4]       # filas 0-4, columnas 1-3

# Selección por etiqueta: loc[filas, columnas]
df.loc[0]               # fila con índice 0
df.loc[:, 'nota_final'] # todas las filas, columna 'nota_final'
df.loc[0:5, ['nombre', 'nota_final']]  # filas 0-5, columnas seleccionadas

Filtros con condiciones booleanas

# Condición simple
aprobados = df[df['nota_final'] >= 4.0]
ingenieria_civil = df[df['carrera'] == 'Ingeniería Civil']

# Múltiples condiciones: & (AND), | (OR), ~ (NOT)
# ⚠️ Usa paréntesis alrededor de cada condición
aprobados_civil = df[
    (df['nota_final'] >= 4.0) &
    (df['carrera'] == 'Ingeniería Civil')
]

buena_nota_o_civil = df[
    (df['nota_final'] >= 6.0) |
    (df['carrera'] == 'Ingeniería Civil')
]

# Excluir una categoría
no_civil = df[df['carrera'] != 'Ingeniería Civil']

# Negar condición compleja
reprobados_no_civil = df[
    ~((df['nota_final'] >= 4.0) | (df['carrera'] == 'Ingeniería Civil'))
]

Filtros con métodos de string

# Columnas de texto: .str accessor
df[df['nombre'].str.contains('García')]            # contiene substring
df[df['nombre'].str.startswith('María')]            # empieza con
df[df['carrera'].str.lower().str.contains('ing')]   # case insensitive

# Regex en filtros de string
df[df['email'].str.match(r'^[\w.-]+@usm\.cl$')]     # solo emails @usm.cl

Filtros con listas: isin()

# Filtrar por múltiples valores (mucho más limpio que OR encadenados)
carreras_ingenieria = ['Ingeniería Civil', 'Ingeniería Industrial',
                        'Ingeniería Eléctrica', 'Ingeniería Informática']

df_ingenieria = df[df['carrera'].isin(carreras_ingenieria)]

# Excluir valores de una lista
df_sin_reprobados_extremos = df[~df['nota_final'].isin([1.0, 1.1, 1.2])]

Filtros con rangos: between()

# Nota entre 4.0 y 5.5 (inclusive en ambos extremos)
df[df['nota_final'].between(4.0, 5.5)]

# Año de ingreso entre 2020 y 2022
df[df['año_ingreso'].between(2020, 2022)]

query(): sintaxis alternativa más legible

# Equivalente a df[(df['nota_final'] >= 4.0) & (df['carrera'] == 'Ingeniería Civil')]
df.query("nota_final >= 4.0 and carrera == 'Ingeniería Civil'")

# Con variables Python en el query (prefijo @)
nota_minima = 4.0
carrera_objetivo = 'Ingeniería Civil'
df.query("nota_final >= @nota_minima and carrera == @carrera_objetivo")

# Muy útil para filtros complejos con múltiples condiciones
df.query("nota_final >= 4.0 and asistencia > 0.75 and año_ingreso >= 2021")

Selección y asignación: loc vs iloc para modificar

# Modificar valores con loc (siempre usa loc/iloc para modificar, nunca encadenes)
df.loc[df['nota_final'] < 1.0, 'nota_final'] = 1.0      # piso de nota en 1.0
df.loc[df['nota_final'] > 7.0, 'nota_final'] = 7.0      # techo de nota en 7.0

# Crear columna nueva basada en condición
df['estado'] = 'Reprobado'
df.loc[df['nota_final'] >= 4.0, 'estado'] = 'Aprobado'

# Con np.where (más conciso para condición binaria)
df['aprobado'] = np.where(df['nota_final'] >= 4.0, True, False)

# Con pd.cut para categorías por rangos
df['categoria_nota'] = pd.cut(
    df['nota_final'],
    bins=[0, 3.9, 4.9, 5.9, 7.0],
    labels=['Reprobado', 'Suficiente', 'Bueno', 'Excelente'],
    right=True
)

Agrupaciones con groupby() y funciones de agregación

La lógica de groupby: Split-Apply-Combine

Agregaciones básicas

# Media de nota_final por carrera
df.groupby('carrera')['nota_final'].mean()

# Múltiples agregaciones en una columna
df.groupby('carrera')['nota_final'].agg(['mean', 'median', 'std', 'count'])

# Múltiples columnas y múltiples funciones
resumen = df.groupby('carrera').agg(
    nota_media    = ('nota_final', 'mean'),
    nota_mediana  = ('nota_final', 'median'),
    nota_min      = ('nota_final', 'min'),
    nota_max      = ('nota_final', 'max'),
    total_alumnos = ('nota_final', 'count'),
    tasa_aprobacion = ('aprobado', 'mean')   # mean de bool = proporción de True
).round(3)

print(resumen)

reset_index() para convertir el resultado en DataFrame limpio

# groupby devuelve el grupo como índice — reset_index() lo convierte en columna
resumen = (
    df.groupby('carrera')['nota_final']
    .mean()
    .reset_index()
    .rename(columns={'nota_final': 'nota_promedio'})
    .sort_values('nota_promedio', ascending=False)
)

Agrupación por múltiples variables

# Promedio por carrera Y año de ingreso
df.groupby(['carrera', 'año_ingreso'])['nota_final'].mean().reset_index()

# Tasa de aprobación por carrera y género
df.groupby(['carrera', 'genero'])['aprobado'].mean().unstack()
# unstack() convierte el segundo nivel del índice en columnas

Funciones de agregación personalizadas

# Función propia con lambda
df.groupby('carrera')['nota_final'].agg(
    lambda x: (x >= 4.0).sum()    # contar aprobados por carrera
)

# Función nombrada
def rango_notas(serie):
    return serie.max() - serie.min()

df.groupby('carrera')['nota_final'].agg(rango_notas)

# Múltiples funciones propias
df.groupby('carrera')['nota_final'].agg(
    aprobados=lambda x: (x >= 4.0).sum(),
    reprobados=lambda x: (x < 4.0).sum(),
    rango=lambda x: x.max() - x.min(),
    p90=lambda x: x.quantile(0.9)
)

transform(): agregar el resultado al DataFrame original

A diferencia de agg() que reduce las filas, transform() devuelve una Serie del mismo tamaño que el DataFrame original — útil para agregar una columna con el promedio del grupo de cada fila:

# Agregar la media del grupo como nueva columna
df['nota_media_carrera'] = df.groupby('carrera')['nota_final'].transform('mean')

# Calcular cuánto se desvía cada estudiante del promedio de su carrera
df['desviacion_del_grupo'] = df['nota_final'] - df['nota_media_carrera']

# Rank dentro del grupo
df['rank_en_carrera'] = df.groupby('carrera')['nota_final'].rank(
    ascending=False, method='min'
)

filter(): filtrar grupos completos

# Conservar solo carreras con más de 50 estudiantes
df_carreras_grandes = df.groupby('carrera').filter(
    lambda grupo: len(grupo) >= 50
)

# Conservar solo carreras con tasa de aprobación > 70%
df_carreras_aprobacion = df.groupby('carrera').filter(
    lambda grupo: (grupo['nota_final'] >= 4.0).mean() > 0.70
)

Tipos de merge (joins)

# DataFrames de ejemplo
df_estudiantes = pd.DataFrame({
    'id_estudiante': [1, 2, 3, 4, 5],
    'nombre': ['Ana', 'Luis', 'María', 'Carlos', 'Sofía'],

df_carreras = pd.DataFrame({
    'id_carrera': [101, 102, 103, 104],
    'nombre_carrera': ['Ingeniería Civil', 'Industrial', 'Eléctrica', 'Informática'],
    'departamento': ['ICM', 'DEI', 'ELO', 'INF']
})

# INNER JOIN (por defecto): solo filas con match en ambas tablas
df_inner = pd.merge(df_estudiantes, df_carreras,
                    on='id_carrera',
                    how='inner')

# LEFT JOIN: todos los estudiantes, con o sin carrera
df_left = pd.merge(df_estudiantes, df_carreras,
                   on='id_carrera',
                   how='left')

# Merge con columnas de distinto nombre
df_merge_cols = pd.merge(
    df_estudiantes,
    df_carreras,
    left_on='id_carrera',
    right_on='id_carrera',
    how='left',
    suffixes=('_est', '_carr')   # sufijos para columnas con mismo nombre
)

Limpieza: valores nulos, duplicados, tipos de dato

Valores nulos (NaN)

# Diagnóstico completo de nulos
def diagnostico_nulos(df):
    """Resumen de valores nulos por columna."""
    nulos = df.isnull().sum()
    pct   = (nulos / len(df) * 100).round(2)
    tipos = df.dtypes

    resumen = pd.DataFrame({
        'nulos': nulos,
        'porcentaje': pct,
        'tipo': tipos
    }).query('nulos > 0').sort_values('porcentaje', ascending=False)

    if len(resumen) == 0:
        print("✅ Sin valores nulos")
    else:
        print(f"⚠️  {len(resumen)} columnas con nulos:\n")
        print(resumen.to_string())
    return resumen

Pipeline de limpieza completo

def limpiar_dataset_notas(df_raw):
    """
    Pipeline de limpieza para dataset de notas institucional.
    Retorna DataFrame limpio con log de cambios.
    """
    df = df_raw.copy()
    log = []

    # 1. Eliminar duplicados exactos
    n_antes = len(df)
    df = df.drop_duplicates()
    log.append(f"Duplicados eliminados: {n_antes - len(df)}")

    # 2. Normalizar tipos
    df['nota_final'] = pd.to_numeric(df['nota_final'], errors='coerce')
    df['fecha']      = pd.to_datetime(df['fecha'], errors='coerce')
    df['carrera']    = df['carrera'].str.strip().str.title()
    log.append("Tipos normalizados")

    # 3. Aplicar límites de notas (escala chilena: 1.0-7.0)
    notas_invalidas = ((df['nota_final'] < 1.0) | (df['nota_final'] > 7.0)).sum()
    df['nota_final'] = df['nota_final'].clip(1.0, 7.0)
    log.append(f"Notas fuera de rango corregidas: {notas_invalidas}")

    # 4. Tratar nulos en columna crítica
    nulos_nota = df['nota_final'].isnull().sum()
    df['nota_final'] = df.groupby('carrera')['nota_final'].transform(
        lambda x: x.fillna(x.median())
    )
    log.append(f"Nulos en nota_final imputados con mediana por carrera: {nulos_nota}")

    # 5. Agregar columna derivada
    df['aprobado'] = df['nota_final'] >= 4.0
    df['categoria_nota'] = pd.cut(
        df['nota_final'],
        bins=[0, 3.9, 4.9, 5.9, 7.01],
        labels=['Reprobado', 'Suficiente', 'Bueno', 'Excelente']
    )

    # Log de cambios
    print("✅ Limpieza completada:")
    for item in log:
        print(f"   • {item}")
    print(f"   • Filas finales: {len(df):,}")

    return df

df_limpio = limpiar_dataset_notas(df_raw)

from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils import get_column_letter

def exportar_informe_formateado(df, ruta, titulo="Informe de Notas"):
    """Exporta DataFrame a Excel con formato básico profesional."""
    with pd.ExcelWriter(ruta, engine='openpyxl') as writer:
        df.to_excel(writer, sheet_name='Datos', index=False, startrow=1)

        ws = writer.sheets['Datos']

        # Título en fila 1
        ws.cell(1, 1, titulo).font = Font(bold=True, size=14)

        # Encabezados en negrita con fondo gris
        for col_idx, col_name in enumerate(df.columns, 1):
            cell = ws.cell(2, col_idx)
            cell.font = Font(bold=True)
            cell.fill = PatternFill("solid", fgColor="CCCCCC")
            cell.alignment = Alignment(horizontal='center')

        # Ajustar ancho de columnas automáticamente
        for col_idx, col in enumerate(df.columns, 1):
            max_len = max(
                len(str(col)),
                df[col].astype(str).str.len().max()
            )
            ws.column_dimensions[get_column_letter(col_idx)].width = min(max_len + 2, 40)

    print(f"✅ Exportado: {ruta}")
