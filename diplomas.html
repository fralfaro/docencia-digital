import unicodedata
from PIL import Image, ImageDraw, ImageFont

# Ruta de la imagen base del diploma
path1 = "diplomas/templates/"
diploma_base = path1 + "ceemat_2025_1semestre.png"

# Lista de nombres de estudiantes
nombres = [
    #"Francisco Alfaro", 
    #"Valeska Canales"
    "Adolfo Seyssel",
    "Alejandro Salas",
    "Andrés Carvajal",
    "Claudio González", 
    "Felipe Asbún",
    "Hans Schulz", 
    "Joaquín Ramírez",
    "Max Rosado",
    "Vicente Moreno", 
    "Pablo Leiva",
    "Javier Diaz", 
    ]

# Cargar la imagen original
img = Image.open(diploma_base)

# Usar una fuente del sistema disponible
font_path = "C:\\Windows\\Fonts\\GARAIT.TTF"
font_size = 70  # Ajustar tamaño según la imagen
font = ImageFont.truetype(font_path, font_size)

# Definir coordenadas aproximadas para el nombre (ajustar según necesidad)
x_pos = 1000  # Ajusta según la imagen
y_pos = 750   # Ajusta según la imagen

def normalizar_nombre(nombre):
    """Convierte el nombre a minúsculas y elimina tildes."""
    nombre = nombre.lower()
    nombre = unicodedata.normalize("NFKD", nombre).encode("ascii", "ignore").decode("utf-8")
    return nombre.replace(" ", "_")

# Generar un diploma por cada estudiante
for nombre in nombres:
    # Crear una copia de la imagen base
    img_edit = img.copy()
    draw = ImageDraw.Draw(img_edit)

    # Insertar el nombre en la imagen (centrado)
    draw.text((x_pos, y_pos), nombre, fill="black", font=font, anchor="mm")  # "mm" centra el texto

    # Normalizar el nombre para el archivo de salida
    nombre_archivo = normalizar_nombre(nombre)

    # Guardar la imagen generada
    path2 = "diplomas/ceemat_2025_1semestre/"
    output_path = path2 + f"{nombre_archivo}.png"
    img_edit.save(output_path)

    print(f"Diploma generado para {nombre}: {output_path}")
